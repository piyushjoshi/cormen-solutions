Pseudo-Code

TREE-INSERT-RECURSIVE (x, z)

x: root node of tree
z: node to insert 

if $z.key \lt x.key$
	if $x.left \ne NIL$
		TREE-INSERT-RECURSIVE (x.left, z)
	else
		z.p = x
		x.left = z
else
	if $x.right \ne NIL$
		TREE-INSERT-RECURSIVE (x.right, z)
	else
		z.p = x
		x.right = z

Javascript Code:

var treeInsertRecursive = function(x, z){
	if(z.key < x.key){
		if(x.left){
			treeInsertRecursive(x.left, z);
		} else{
			z.p = x;
			x.left = z;
		}
	} else{
		if(x.right){
			treeInsertRecursive(x.left, z);
		} else{
			z.p = x;
			x.right = z;
		}
	}
};
